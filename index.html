async function descargarDXF() { if(dxfVertices.length < 3) return; try { const respuesta = await fetch(`${BACKEND_URL}/api/generar-dxf`, { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ vertices: dxfVertices, radioCustom: parseFloat(document.getElementById('dxfRadioVertices').value) || 1, txtVerticesCustom: parseFloat(document.getElementById('dxfTxtVertices').value) || 1.5, txtTablaCustom: parseFloat(document.getElementById('dxfTxtTabla').value) || 0.8 }) }); const datos = await respuesta.json(); // Descarga el archivo DXF que viene procesado del servidor triggerDescarga(datos.dxfContenido, 'poligono_curvas_procesado.dxf', 'application/dxf'); // Descarga la minuta llamando a TU función original de letras para DXF if(document.getElementById('dxfGenerarTextoDoc').checked) { // Llamamos a tu algoritmo nativo que maneja curvas y letras const textoValidoDXF = dxfGenerarTextoDescriptivo(); // Forzamos el charset-utf-8 en el trigger para evitar la ventana de conversión triggerDescarga(textoValidoDXF, 'minuta_dxf.doc', 'application/msword;charset=utf-8'); } } catch (err) { alert("Error al procesar la descarga desde el servidor."); } }
